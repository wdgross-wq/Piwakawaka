            startPlugin() {
                this.actionIntervalId = setInterval(this.decideNextAction.bind(this), 1500);
                this.clearPoses(); 
                this.y = window.innerHeight / 2; 
                this.takeOff();
                this.wander();
            }

            clearPoses() {
                this.container.classList.remove('pose-h-alert', 'pose-h-rest', 'pose-h-curious', 'pose-h-display', 'pose-v-cling', 'pose-h-clean', 'pose-stretch', 'pwk-fanned', 'pwk-fast');
                this.currentPose = '';
            }

            setAnimationState(newState) {
                this.state = newState;
                this.container.classList.remove('state-fluttering', 'state-diving');
                
                if (newState !== 'flying' && newState !== 'perched' && newState !== 'hidden') {
                    this.container.classList.add(`state-${newState}`);
                }
                
                if (newState === 'perched') {
                    this.container.classList.add('state-perched');
                    this.applyRandomPerchPose();
                } else {
                    this.container.classList.remove('state-perched');
                    this.clearPoses();
                }
            }

            applyRandomPerchPose() {
                if (this.isSequencing) return; 
                
                this.clearPoses();
                if (this.perchSide === 'top') {
                    const poses = ['pose-h-alert', 'pose-h-rest', 'pose-h-curious', 'pose-h-alert', 'pose-h-display', 'pose-h-clean'];
                    this.currentPose = poses[Math.floor(Math.random() * poses.length)];
                    this.container.classList.add(this.currentPose);
                    
                    if (this.currentPose === 'pose-h-display' || this.currentPose === 'pose-h-clean') {
                        this.container.classList.add('pwk-fanned');
                    } else {
                        this.container.classList.add('pwk-closed');
                    }
                } else {
                    this.currentPose = 'pose-v-cling';
                    this.container.classList.add(this.currentPose, 'pwk-fanned');
                }
            }

            decideNextAction() {
                if (this.state === 'hidden' || this.isSequencing || this.isJumping) return;
                
                if (this.state === 'perched') {
                    const rand = Math.random();
                    if (rand < 0.15) {
                        this.takeOff();
                        this.wander();
                    } else if (rand < 0.35) {
                        this.hopToNewPerch();
                    } else {
                        if (this.perchSide === 'top' && Math.random() < 0.4) {
                            this.facingRight = !this.facingRight; 
                        }
                        
                        const actionRand = Math.random();
                        if (actionRand < 0.5) {
                            if (this.perchSide === 'top' && Math.random() < 0.5) {
                                this.applyRandomPerchPose();
                            } else {
                                this.container.classList.add('pwk-fidget', 'pwk-fanned'); 
                                this.container.classList.remove('pwk-closed');
                                setTimeout(() => {
                                    if (this.state === 'perched' && !this.isSequencing) {
                                        this.container.classList.remove('pwk-fidget');
                                        if (this.currentPose !== 'pose-h-display' && this.currentPose !== 'pose-v-cling' && this.currentPose !== 'pose-h-clean') {
                                            this.container.classList.remove('pwk-fanned'); 
                                            this.container.classList.add('pwk-closed');
                                        }
                                    }
                                }, 200); 
                            }
                        } else if (actionRand < 0.8) {
                            this.container.classList.add('pwk-head-bob');
                            setTimeout(() => { this.container.classList.remove('pwk-head-bob'); }, 300);
                        }
                    }
                } else {
                    if (Math.random() < 0.8) this.seekPerch(); else this.wander();
                }
            }

            takeOff() { 
                this.setAnimationState('flying'); 
                this.perchElement = null; 
                this.vy = -10; 
                this.wander(); 
            }

            hopToNewPerch() {
                this.setAnimationState('flying');
                this.vy = -7; 
                this.vx = this.facingRight ? 10 : -10;
                this.seekPerch();
            }

            scare() {
                if (this.state === 'hidden' || this.isSequencing) return;
                this.takeOff();
                this.vx = this.facingRight ? -25 : 25; 
                this.wander();
            }

            // ==========================================
            // KINEMATIC JUMP & TREE SEQUENCES
            // ==========================================
            startKinematicJump(endX, endY, height, duration, callback) {
                this.isJumping = true;
                this.jumpStartX = this.x;
                this.jumpStartY = this.y;
                this.jumpEndX = endX;
                this.jumpEndY = endY;
                this.jumpHeight = height; // Negative arcs UP
                this.jumpDuration = duration;
                this.jumpTime = 0;
                this.jumpCallback = callback;
                
                this.setAnimationState('flying');
                this.container.classList.add('pwk-fanned', 'pwk-closed'); 
            }

            hideInTree() {
                if (this.state === 'hidden' || this.isSequencing) return;
                this.isSequencing = true;
                
                this.treeBg.style.opacity = '1';
                this.treeFg.style.opacity = '1';
                
                this.perchElement = document.getElementById('pwk-tree-branch-target');
                this.perchSide = 'top';
                this.perchOffset = 20; 
                this.setAnimationState('flying');
                this.updatePerchTarget();
            }

            emergeFromTree() {
                if (this.state !== 'hidden') return;
                this.isSequencing = true;
                
                this.treeBg.style.opacity = '1';
                this.treeFg.style.opacity = '1';
                
                const hole = document.getElementById('pwk-tree-hole').getBoundingClientRect();
                const branch = document.getElementById('pwk-tree-branch-target').getBoundingClientRect();
                
                this.x = hole.left + hole.width / 2;
                this.y = hole.top + hole.height / 2 + 30; // Start deep
                this.container.style.opacity = '1';
                this.state = 'flying'; 
                this.facingRight = false; 
                
                // Jump OUT of hole onto branch
                this.startKinematicJump(
                    branch.left + branch.width / 2, 
                    branch.top - 22,  
                    -60, // Arc High
                    0.5, 
                    () => {
                        this.setAnimationState('perched');
                        this.vx = 0; this.vy = 0;
                        
                        // STRETCH SEQUENCE
                        this.clearPoses();
                        this.container.classList.add('pose-stretch', 'pwk-fanned');
                        
                        setTimeout(() => {
                            this.isSequencing = false;
                            this.clearPoses();
                            this.takeOff();
                            
                            setTimeout(() => {
                                this.treeBg.style.opacity = '0';
                                this.treeFg.style.opacity = '0';
                            }, 800);
                        }, 1200); 
                    }
                );
            }

            // ==========================================
            // PHYSICS & BOUNDS
            // ==========================================
            clampPosition() {
                if (isNaN(this.x)) this.x = window.innerWidth / 2; if (isNaN(this.y)) this.y = window.innerHeight / 2;
                this.x = Math.max(60, Math.min(window.innerWidth - 60, this.x));
                this.y = Math.max(60, Math.min(window.innerHeight - 60, this.y));
            }

            wander() {
                const pad = 100; 
                this.targetX = pad + Math.random() * (window.innerWidth - pad * 2);
                this.targetY = pad + Math.random() * (window.innerHeight - pad * 2);
                this.perchElement = null;
            }

            seekPerch() {
                if (this.isSequencing) return;
                
                const candidates = Array.from(document.querySelectorAll(this.perchSelector)).filter(el => {
                    const rect = el.getBoundingClientRect();
                    return rect.width > 0 && rect.height > 0 && 
                           rect.top > 60 && rect.left > 60 && 
                           rect.bottom < window.innerHeight - 60 && rect.right < window.innerWidth - 60;
                });

                if (candidates.length > 0) {
                    this.perchElement = candidates[Math.floor(Math.random() * candidates.length)];
                    const rect = this.perchElement.getBoundingClientRect();
                    const rand = Math.random(); 
                    this.perchSide = rand < 0.5 ? 'top' : (rand < 0.75 ? 'left' : 'right');
                    this.perchOffset = Math.random() * (this.perchSide === 'top' ? rect.width : rect.height); 
                    this.updatePerchTarget();
                } else this.wander();
            }

            updatePerchTarget() {
                if (!this.perchElement) return;
                const rect = this.perchElement.getBoundingClientRect();
                
                // Offset perfectly to rest exact claws on edges
                if (this.perchSide === 'top') { this.targetX = rect.left + this.perchOffset; this.targetY = rect.top - 22; } 
                else if (this.perchSide === 'left') { this.targetX = rect.left - 18; this.targetY = rect.top + this.perchOffset; } 
                else if (this.perchSide === 'right') { this.targetX = rect.right + 18; this.targetY = rect.top + this.perchOffset; }
            }

            updateTransform() {
                if (this.state === 'hidden') return;
                let angle = 0;
                
                if (this.state === 'perched') {
                    if (this.perchSide === 'left') angle = -80; 
                    else if (this.perchSide === 'right') angle = -80; 
                } 
                else if (this.isJumping) {
                    // Track pitch accurately during the parabola
                    angle = this.currentVy > 0 ? 60 : -35; 
                    if (!this.facingRight) angle *= -1;
                }
                else {
                    let velAngle = Math.atan2(this.vy, Math.abs(this.vx)) * (180 / Math.PI);
                    angle = Math.max(-50, Math.min(70, velAngle)); // Impossible to fall bottom-first
                    
                    if (this.container.classList.contains('pwk-fanned') && this.vy > -3) angle = -35; 
                    else if (this.container.classList.contains('state-diving')) angle = 75; 
                    
                    if (!this.facingRight) angle *= -1;
                }

                if (this.state === 'perched' && this.perchSide === 'right' && !this.facingRight) angle = -80; 
                else if (this.state === 'perched' && this.perchSide === 'top' && !this.facingRight) angle *= -1; 

                const scaleX = this.facingRight ? 1 : -1;
                this.container.style.transform = `translate(${this.x}px, ${this.y}px)`;
                this.sprite.style.transform = `scaleX(${scaleX}) rotate(${angle}deg) scale(0.85)`;
            }

            update(timestamp) {
                const dt = Math.min((timestamp - this.lastTime) / 1000, 0.05); 
                this.lastTime = timestamp; 
                this.lastY = this.y; // For tracking instantaneous velocity

                if (this.state === 'hidden') {
                    // Do nothing
                }
                else if (this.isJumping) {
                    this.jumpTime += dt;
                    let t = Math.min(this.jumpTime / this.jumpDuration, 1.0);
                    
                    this.x = this.jumpStartX + (this.jumpEndX - this.jumpStartX) * t;
                    this.y = this.jumpStartY + (this.jumpEndY - this.jumpStartY) * t + (4 * this.jumpHeight * t * (1 - t));
                    this.currentVy = this.y - this.lastY; // Used for pitch calculation
                    
                    if (t >= 1.0) {
                        this.isJumping = false;
                        if (this.jumpCallback) this.jumpCallback();
                    }
                }
                else if (this.state === 'perched' && this.perchElement) {
                    const rect = this.perchElement.getBoundingClientRect();
                    if (this.perchSide === 'top') { this.x = rect.left + this.perchOffset; this.y = rect.top - 22; } 
                    else if (this.perchSide === 'left') { this.x = rect.left - 18; this.y = rect.top + this.perchOffset; } 
                    else if (this.perchSide === 'right') { this.x = rect.right + 18; this.y = rect.top + this.perchOffset; }
                    
                    if (rect.bottom < 60 || rect.top > window.innerHeight - 60) {
                        this.takeOff();
                    }
                } 
                else {
                    if (this.perchElement) this.updatePerchTarget();
                    let dx = this.targetX - this.x; let dy = this.targetY - this.y;
                    let dist = Math.max(0.1, Math.sqrt(dx*dx + dy*dy));

                    if (this.perchElement && dist < 20) {
                        
                        if (this.isSequencing && this.perchElement.id === 'pwk-tree-branch-target') {
                            this.setAnimationState('perched'); this.vx = 0; this.vy = 0;
                            this.facingRight = false; 
                            this.clearPoses();
                            
                            // "Coast is Clear" sequence
                            this.container.classList.add('pose-h-alert', 'pwk-closed');
                            
                            setTimeout(() => {
                                if(!this.isSequencing) return;
                                this.facingRight = true; 
                                this.container.classList.remove('pose-h-alert');
                                this.container.classList.add('pose-h-curious'); // Lean to look
                                
                                setTimeout(() => {
                                    if(!this.isSequencing) return;
                                    const hole = document.getElementById('pwk-tree-hole').getBoundingClientRect();
                                    
                                    this.startKinematicJump(
                                        hole.left + hole.width / 2, 
                                        hole.top + hole.height / 2 + 30, // dive deep into hole
                                        -35, 
                                        0.4, 
                                        () => {
                                            this.state = 'hidden';
                                            this.container.style.opacity = '0';
                                            setTimeout(() => {
                                                this.treeBg.style.opacity = '0';
                                                this.treeFg.style.opacity = '0';
                                            }, 500);
                                        }
                                    );
                                }, 500);
                            }, 600);
                        } 
                        else if (!this.isSequencing) {
                            this.setAnimationState('perched'); this.vx = 0; this.vy = 0;
                            if (this.perchSide === 'left') this.facingRight = true;
                            else if (this.perchSide === 'right') this.facingRight = false;
                            else this.facingRight = this.x < this.perchElement.getBoundingClientRect().left + this.perchElement.getBoundingClientRect().width / 2;
                            
                            this.container.classList.remove('pwk-fanned', 'state-fluttering', 'state-diving');
                            this.container.classList.add('pwk-closed');
                        }
                    } else {
                        let accel = 1.0;
                        this.vx += (dx / dist) * accel; this.vy += (dy / dist) * accel;
                        
                        this.flightTimer += dt * 12; this.vy += Math.sin(this.flightTimer) * 0.8; 

                        this.dartTimer -= dt;
                        if (this.dartTimer <= 0 && dist > 150 && !this.isSequencing) {
                            this.dartTimer = 3 + Math.random() * 4; 
                            this.vx += (Math.random() - 0.5) * 20; this.vy += (Math.random() - 0.5) * 20 - 4; 
                            this.container.classList.add('pwk-fanned'); this.container.classList.remove('pwk-closed');
                            setTimeout(() => {
                                if (this.state === 'flying' && this.vy < 3) {
                                    this.container.classList.remove('pwk-fanned'); this.container.classList.add('pwk-closed');
                                }
                            }, 350);
                        }

                        this.container.classList.remove('state-diving', 'state-fluttering');
                        
                        if (this.perchElement && dist < 150) {
                            this.container.classList.add('state-fluttering', 'pwk-fanned'); this.container.classList.remove('pwk-closed');
                            this.vx *= 0.85; this.vy *= 0.85;
                        } else if (this.vy > 6) {
                            this.container.classList.add('state-diving', 'pwk-closed'); this.container.classList.remove('pwk-fanned');
                            this.vy += 0.3; 
                        } else {
                            this.container.classList.add('state-fluttering');
                            let currentAngle = Math.atan2(this.vy, this.vx); let targetAngle = Math.atan2(dy, dx);
                            let angleDiff = Math.abs(currentAngle - targetAngle); const vel = Math.sqrt(this.vx*this.vx + this.vy*this.vy);
                            
                            if (angleDiff > 1.2 || vel < 5) {
                                this.container.classList.add('pwk-fanned'); this.container.classList.remove('pwk-closed');
                            } else if (this.dartTimer < this.dartTimer - 0.35) {
                                this.container.classList.add('pwk-closed'); this.container.classList.remove('pwk-fanned');
                            }
                            
                            if (vel > 10 && !this.container.classList.contains('state-fluttering')) {
                                this.container.classList.add('pwk-fast');
                            } else {
                                this.container.classList.remove('pwk-fast');
                            }
                        }
                        this.vx *= 0.92; this.vy *= 0.92;
                        
                        this.x += this.vx * (dt * 60); this.y += this.vy * (dt * 60);
                        if (Math.abs(this.vx) > 1.0) this.facingRight = this.vx > 0;
                    }
                }

                if (!this.isJumping) this.clampPosition();
                this.updateTransform();
                requestAnimationFrame(this.update.bind(this));
            }
        }

        window.addEventListener('DOMContentLoaded', () => {
            window.piwakawaka = new PiwakawakaPlugin({ perchSelector: '.bird-perch, nav a, button' });

            const hideBtn = document.getElementById('hide-btn');
            const showBtn = document.getElementById('show-btn');

            hideBtn.addEventListener('click', () => {
                window.piwakawaka.hideInTree();
                hideBtn.classList.add('opacity-50', 'pointer-events-none');
                setTimeout(() => showBtn.classList.remove('opacity-50', 'pointer-events-none'), 3000);
            });

            showBtn.addEventListener('click', () => {
                window.piwakawaka.emergeFromTree();
                showBtn.classList.add('opacity-50', 'pointer-events-none');
                setTimeout(() => hideBtn.classList.remove('opacity-50', 'pointer-events-none'), 3000);
            });

            document.getElementById('scare-btn').addEventListener('click', () => {
                window.piwakawaka.scare();
            });
        });
    </script>
